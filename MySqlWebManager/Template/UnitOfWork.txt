using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
{0}
{1}
{
    public class UnitOfWork : IUnitOfWork, IDisposable
    {
        private DBEntities db;
        public DBEntities DB { get { return db = (db ?? (new DBEntities())); } }

        public void Save()
        {
            this.db.SaveChanges();
        }

        private bool hasDisposed = false;
        public void Dispose()
        {
            Dispose(true);
            GC.SuppressFinalize(this);
        }
        protected virtual void Dispose(bool disposing)
        {
            if (!hasDisposed)
            {
                if (disposing)
                {
                    if (db != null) db.Dispose();
                }
                hasDisposed = true;
            }
        }
        ~UnitOfWork()
        {
            Dispose(false);
        }
    }
}
